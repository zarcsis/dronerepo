Setup

curl -fsSL https://zarcsis.github.io/dronerepo/repo.key | sudo tee /etc/apt/trusted.gpg.d/dronerepo.asc > /dev/null
echo "deb https://zarcsis.github.io/dronerepo/ $(lsb_release -sc) main" | sudo tee /etc/apt/sources.list.d/dronerepo.list
sudo apt update
